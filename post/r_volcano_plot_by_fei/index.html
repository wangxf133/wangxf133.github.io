4.数据处理

data$fac=as.factor((data$ratio_116vs117 > 1.2 | data$ratio_116vs117 < 0.8333)&data$qvalue_116vs117 < 0.05)
data$fac

5.简单绘图

ggplot(data,aes(x=log2(ratio_116vs117) , y=-log10(qvalue_116vs117) , color= fac))+geom_point()

6.标记significant(从0列开始，正是因为附在后面，前面还多加了一列fac，所以是第15列)

data$significant <- "no"
up_row <- which(data$ratio_116vs117 > 1.2 & data$qvalue_116vs117 < 0.05)
data[up_row,15] <- "up"

down_row <- which(data$ratio_116vs117 < 0.8333 & data$qvalue_116vs117 < 0.05)
data[down_row,15] <- "down"

data$significant

7.绘图——区分significant---up and down

ggplot(data,aes(x=log2(ratio_116vs117) , y=-log10(qvalue_116vs117) , color= significant))+geom_point()

8.美化一下

ggplot(data,aes(x=log2(ratio_116vs117) , y=-log10(qvalue_116vs117) , color= significant)) + geom_point() + scale_color_manual(values = c("goldenrod1","lightgrey","firebrick2")) +
  theme_bw() +
  theme(panel.grid.major = element_blank(), panel.grid.minor = element_blank(),legend.key=element_blank())

9.生成图片和需要的文件

png(file = "ETWvsTD265(116vs117)_volcano_significant.png")
